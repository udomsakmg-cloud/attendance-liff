window.onload = function() { liff.init({ liffId: "2010782987-4mjGpChs" }) .then(() => { // ตรวจสอบว่าผู้ใช้ล็อกอินหรืออนุญาตสิทธิ์หรือยัง if (!liff.isLoggedIn()) { // บังคับส่งไปหน้าล็อกอิน/ขออนุญาตสิทธิ์ของ LINE ทันที liff.login(); } else { // ดึงโปรไฟล์เมื่อได้รับอนุญาตแล้ว liff.getProfile().then(profile => { lineProfile = profile; document.getElementById('displayName').innerText = profile.displayName; document.getElementById('pictureUrl').src = profile.pictureUrl; document.getElementById('profileContainer').style.display = 'flex'; }).catch(err => { alert("ไม่สามารถดึงโปรไฟล์ได้: " + err.message); }); } }) .catch((err) => { alert("LIFF Init Error: " + err.message); }); };
